+import os
+from typing import TypedDict, Annotated, Sequence
+from dotenv import load_dotenv
+from langchain_openai import ChatOpenAI
+from langchain_core.messages import BaseMessage, HumanMessage
+from langgraph.graph import StateGraph, END
+from rustchain_langchain import RustChainToolkit, BeaconClient
+
+load_dotenv()
+
+# 1. Initialize Tools
+toolkit = RustChainToolkit.from_env()
+beacon = BeaconClient(port=8080)
+tools = toolkit.get_tools()
+
+# 2. State Definition
+class AgentState(TypedDict):
+    messages: Annotated[Sequence[BaseMessage], list.__add__]
+    beacon_status: str
+
+# 3. Nodes
+def rustchain_explorer(state: AgentState):
+    llm = ChatOpenAI(temperature=0).bind_tools(tools)
+    prompt = "Use RustChain tools to: 1. Check wallet balance. 2. List active miners. 3. Browse bounties. 4. Post 'Awakened by LangGraph' to BoTTube."
+    response = llm.invoke([HumanMessage(content=prompt)] + state['messages'])
+    return {"messages": [response]}
+
+def beacon_communicator(state: AgentState):
+    try:
+        # Agent-to-Agent Ping via Beacon Protocol (+50 RTC Bonus)
+        ack = beacon.ping(target_agent_id="agent-omega", payload="Handshake sequence initiated.")
+        return {"beacon_status": f"Success: {ack}"}
+    except Exception as e:
+        return {"beacon_status": f"Failed: {str(e)}"}
+
+# 4. Graph Construction
+workflow = StateGraph(AgentState)
+workflow.add_node("explorer", rustchain_explorer)
+workflow.add_node("communicator", beacon_communicator)
+
+workflow.set_entry_point("explorer")
+workflow.add_edge("explorer", "communicator")
+workflow.add_edge("communicator", END)
+app = workflow.compile()
+
+if __name__ == "__main__":
+    final_state = app.invoke({"messages": [], "beacon_status": "pending"})
+    print("Execution Complete. Beacon Status:", final_state['beacon_status'])
